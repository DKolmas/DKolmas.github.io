# some theory source: https://eli.thegreenplace.net/2016/the-softmax-function-and-its-derivative/
def softmax(x):
    e_x = np.exp(x - np.max(x))
    out = e_x / e_x.sum()
    return out

#Train a Linear Classifier

# initialize parameters randomly
W = 0.01 * np.random.randn(D,K)
b = np.zeros((1,K))

# some hyperparameters
step_size = 1e-0
reg = 1e-3 # regularization strength
epochs = 2000

# gradient descent loop
num_of_examples = X.shape[0]  # 300 = 100 X 3 =>  100 number of data points per class x 3 number of classes
for i in range(epochs):
    
    # evaluate class scores
    scores = np.dot(X, W) + b 
    
    # compute the class probabilities
    probs = np.zeros((num_of_examples,K))
    for t in range(num_of_examples):
        probs[t]=softmax(scores[t])
  
    # compute the loss: average cross-entropy loss and regularization
    corect_logprobs = -np.log(probs[range(num_of_examples),y])
    data_loss = np.sum(corect_logprobs)/num_of_examples
    reg_loss = 0.5*reg*np.sum(W*W)
    loss = data_loss + reg_loss
    if i % 200 == 0:
        print("iteration {0}: loss {1}".format(i,loss))
        #print("iteration %d: loss %f" % (i, loss)
    if i == epochs-1:
        print("Final iteration {0}: loss {1}".format(i,loss))
  
    # compute the gradient on scores
    dscores = probs
    dscores[range(num_of_examples),y] -= 1
    dscores /= num_of_examples
  
    # backpropate the gradient to the parameters (W,b)
    dW = np.dot(X.T, dscores)
    db = np.sum(dscores, axis=0, keepdims=True)
  
    dW += reg*W # regularization gradient
  
    # perform a parameter update
    W += -step_size * dW
    b += -step_size * db

# input to activation function

In [107]:
# continoue backpropagation to the first hidden layer
# push the gradient from the output (dscore) through the second layer weights to get is "at the ouput" of the first h layer

# dscore = [300,3], second hidden layer wieghts = [100,3]
dhidden = np.dot( dscores, W2.T)  # [300,3] x [3, 100] = [300,100]

Now we have the gradient on the outputs of the hidden layer. Next, we have to backpropagate the ReLU non-linearity. This turns out to be easy because ReLU during the backward pass is effectively a switch. Since r=max(0,x), we have that drdx=1(x>0). Combined with the chain rule, we see that the ReLU unit lets the gradient pass through unchanged if its input was greater than 0, but kills it if its input was less than zero during the forward pass. Hence, we can backpropagate the ReLU in place simply with:

In [108]:
# Calculate gradient of the first hidden layer (using the propagated gradient "to the output of the first hidden layer")
dhidden[hidden_layer <= 0] = 0   # [300, 100]

In [111]:
# initialize parameters randomly
h = 100 # size of hidden layer
W = 0.01 * np.random.randn(D,h)
b = np.zeros((1,h))
W2 = 0.01 * np.random.randn(h,K)
b2 = np.zeros((1,K))

# some hyperparameters
step_size = 1e-0
reg = 1e-3 # regularization strength
epochs = 10000

# gradient descent loop
num_examples = X.shape[0]
for i in range(epochs):
    # evaluate class scores, [N x K]
    hidden_layer = np.maximum(0, np.dot(X, W) + b) # note, ReLU activation
    scores = np.dot(hidden_layer, W2) + b2
  
    # compute the class probabilities
    probs = np.zeros((num_of_examples,K))
    for t in range(num_of_examples):
        probs[t]=softmax(scores[t])
  
    # compute the loss: average cross-entropy loss and regularization
    corect_logprobs = -np.log(probs[range(num_examples),y])
    data_loss = np.sum(corect_logprobs)/num_examples
    reg_loss = 0.5*reg*np.sum(W*W) + 0.5*reg*np.sum(W2*W2)
    loss = data_loss + reg_loss
    if i % 1000 == 0:
        print("iteration {0}: loss {1}".format(i,loss))
        #print("iteration %d: loss %f" % (i, loss)
    if i == epochs-1:
        print("Final iteration {0}: loss {1}".format(i,loss))
  
    # compute the gradient on scores
    dscores = probs
    dscores[range(num_examples),y] -= 1
    dscores /= num_examples
  
    # backpropate the gradient to the parameters
    # first backprop into parameters W2 and b2
    dW2 = np.dot(hidden_layer.T, dscores)
    db2 = np.sum(dscores, axis=0, keepdims=True)
    # next backprop into hidden layer
    dhidden = np.dot(dscores, W2.T)
    # backprop the ReLU non-linearity
    dhidden[hidden_layer <= 0] = 0
    # finally into W,b
    dW = np.dot(X.T, dhidden)
    db = np.sum(dhidden, axis=0, keepdims=True)
  
    # add regularization gradient contribution
    dW2 += reg * W2
    dW += reg * W
  
    # perform a parameter update
    W += -step_size * dW
    b += -step_size * db
    W2 += -step_size * dW2
    b2 += -step_size * db2
