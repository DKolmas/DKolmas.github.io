def Q_learninig_FrozeLake( InParam ):
    start = time.time()
    
    # Initialize Q table
    #Q = np.zeros([env.observation_space.n, env.action_space.n])
    Q = np.random.randn(env.observation_space.n, env.action_space.n)
    #Q = np.random.uniform(0,1,(env.observation_space.n, env.action_space.n))
    
    revPerEpisode = list() # One reward is calculatd at the end of each episode
    avRewInSteps = list()  # For specific episode number (see: InParam['recStep']) average reward 
                           # up to the time is calculated
    actionInStep = list()
    stepsInEpisode = list()
    
    eta = InParam['eta']
    gma = InParam['gma']
    epis = InParam['epis']
    recordSteps = InParam['recStep']
    print("Number of episodes: ",epis)
        
    for episode in range(epis):
        s = env.reset()  # Reset environment
        rAll = 0
        steps = 0
        done = False 
        actionInStep.extend([5]) # To distinguish one episode from another
        #The Q-Table learning algorithm
        while not done:
            steps += 1
            # Choose action from Q table
            # Q table is perturbated with random values (normal, mean=0, std=1)
            # whose impact is reduced with episode number (i index)
            a = np.argmax(Q[s,:] + np.random.randn(1,env.action_space.n)*(1./(episode+1)))
            #a = np.argmax(Q[s,:])
            actionInStep.extend([a])
            s_before = s
            # Get new state, new reward and indication if episode came to the ned (done) from environment.
            # Ignore debug info (fourth argument)
            s1, r1, done, _ = env.step(a)
            # NOTE: I use the notation of reward from the state s as r1 (nor r). 
            # It is due to Sutton book: Reinforcement Learning: an Introduction
            # Update Q-Table with new knowledge
            Q[s,a] = Q[s,a] + eta*(r1 + gma*np.max(Q[s1,:]) - Q[s,a])        
            rAll += r1
            s = s1
            if done == True:
                stepsInEpisode.extend([steps])
                break
        revPerEpisode.extend([rAll])
        
        if episode in recordSteps:
            avRewInSteps.extend([ (episode, sum(revPerEpisode)/episode) ])
        
        # If activated through parameter msgPrint 
        if episode > (epis-5):
            if InParam['msgPrint']:
                env.render()
                print("Reward value: ",r1," state before: ",s_before," action taken: ",a)
                print(" ")
    
    avRewInSteps.extend([ (epis, np.sum(revPerEpisode)/epis) ])
    print("Averaged reward overe all episodes ",avRewInSteps[-1], " Average steps/epis ",np.sum(stepsInEpisode)/epis)
    end = time.time()
    print("Elapsed time: ", (end-start))
    print(" ")
    return avRewInSteps, actionInStep
