def Q_learninig_FrozeLake( InParam ):
    start = time.time()
    
    # Initialize Q table
    Q = np.zeros([env.observation_space.n, env.action_space.n])
    #Q = np.random.randn(env.observation_space.n, env.action_space.n)
    #Q = np.random.uniform(0,1,(env.observation_space.n, env.action_space.n))
    
    rewPerEpisode = list() # One reward is calculatd at the end of each episode
    avRew = list()         # For specific episode number (see: InParam['recStep']) average reward 
                           # up to the episode is calculated
    stepsInEpisode = list()# For each episode number of steps that were take is recoded. Typically 
                           # small number of steps sugest an agent fails quickly
    
    l_rate = InParam['l_rate']
    gma = InParam['gma']
    epis = InParam['episodes']
    recordSteps = InParam['recordSteps']
    print("Number of episodes: ",epis)
    actionHist = dict()
    actionHist['action0'] = list()
    actionHist['action1'] = list()
    actionHist['action2'] = list()
    actionHist['action3'] = list()
    
    epsilon = lambda currentEpisode: (-0.9/(0.8*epis))*currentEpisode + 1

    for episode in range(epis):
        actionCount = np.zeros((4,1))
        s = env.reset()  # Reset environment
        rAll = 0
        steps = 0
        done = False 

        while not done:
            steps += 1
            # Choose action from Q table
            if InParam['actionType'] == 'perturbated':
                a = np.argmax(Q[s,:] + np.random.randn(1,env.action_space.n)*(1./(episode+1)))
            elif InParam['actionType'] == 'epsilon':
                if np.random.uniform(0, 1) < max( epsilon( episode ),0.1, ):
                    a = env.action_space.sample()
                else:
                    a = np.argmax(Q[s,:])
            
            actionCount[a,0] += 1
            s1, r1, done, _ = env.step(a)
            Q[s,a] = Q[s,a] + l_rate*(r1 + gma*np.max(Q[s1,:]) - Q[s,a])    
            rAll += r1
            s_before = s
            s = s1
            if done == True:
                stepsInEpisode.extend([steps])
                break
        rewPerEpisode.extend([rAll])

        actionHist['action0'].extend([actionCount[0,0]])
        actionHist['action1'].extend([actionCount[1,0]])
        actionHist['action2'].extend([actionCount[2,0]])
        actionHist['action3'].extend([actionCount[3,0]])
        
        if episode in recordSteps:
            avRew.extend([ (episode, np.sum(rewPerEpisode)/episode) ])
        
        # If activated through parameter msgPrint 
        if episode > (epis-5):
            if InParam['msgPrint']:
                env.render()
                print("Reward value: ",r1," state before: ",s_before," action taken: ",a,1)
                print(" ")
    
    avRew.extend([ (epis, np.sum(rewPerEpisode)/epis) ])
    print("Averaged reward overe all episodes ",avRew[-1], \
          " Average steps/epis ",np.round(np.sum(stepsInEpisode)/epis,1))
    end = time.time()
    print("Elapsed time: ", np.round((end-start),2),'seconds')
    print(" ")
    return avRew, stepsInEpisode, actionHist

def plotProgress( avRewIn, stepsInEpIn, windowSize=50):
    f = plt.figure(figsize=(14,3))
    ax1 = f.add_subplot(121)
    ax2 = f.add_subplot(122)
    # https://stackoverflow.com/questions/13728392/moving-average-or-running-mean
    avSteps = np.convolve(stepsInEpIn, np.ones((windowSize,))/windowSize, mode='valid')
    p1 = ax1.plot(avSteps)
    p1 = ax1.set(xlabel='Episode', ylabel='Steps in one episode', title='Averaged number of steps in episode')
    p2 = ax2.plot([elm[0] for elm in avRewIn],[elm[1] for elm in avRewIn])
    p2 = ax2.set(xlabel='Episode', ylabel='Average reward', title='Average reward calculated from begining')
    return
